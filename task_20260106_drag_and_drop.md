function pointIndex = findClosestPoint(app, clickPosition)
    % FINDCLOSESTPOINT –ù–∞–π—Ç–∏ –±–ª–∏–∂–∞–π—à—É—é —Ç–æ—á–∫—É –∫ –ø–æ–∑–∏—Ü–∏–∏ –∫–ª–∏–∫–∞
    %   POINTINDEX = FINDCLOSESTPOINT(APP, CLICKPOSITION)
    %   –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç [curveIndex, pointIndex] –±–ª–∏–∂–∞–π—à–µ–π —Ç–æ—á–∫–∏
    
    % –ü–æ–ª—É—á–∏—Ç—å –≤—Å–µ –ª–∏–Ω–∏–∏ –Ω–∞ –≥—Ä–∞—Ñ–∏–∫–µ
    children = app.axPlot.Children;
    
    minDistance = inf;
    pointIndex = [];
    
    % –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –∫–∞–∂–¥—É—é –ª–∏–Ω–∏—é (–∫—Ä–∏–≤—É—é)
    for curveIdx = 1:length(children)
        if isa(children(curveIdx), 'matlab.graphics.chart.primitive.Line')
            line = children(curveIdx);
            xData = line.XData;
            yData = line.YData;
            
            % –í—ã—á–∏—Å–ª–∏—Ç—å —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –¥–æ –≤—Å–µ—Ö —Ç–æ—á–µ–∫ —ç—Ç–æ–π –∫—Ä–∏–≤–æ–π
            for pointIdx = 1:length(xData)
                point = [xData(pointIdx), yData(pointIdx)];
                distance = norm(clickPosition(1:2) - point);
                
                if distance < minDistance
                    minDistance = distance;
                    pointIndex = [curveIdx, pointIdx];
                end
            end
        end
    end
    
    % –ü—Ä–æ–≤–µ—Ä–∏—Ç—å, —á—Ç–æ —Ç–æ—á–∫–∞ –Ω–∞–π–¥–µ–Ω–∞ –∏ –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –±–ª–∏–∑–∫–æ
    if isempty(pointIndex) || minDistance > 0.1 % –ü–æ—Ä–æ–≥ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è
        pointIndex = [];
    end
end
